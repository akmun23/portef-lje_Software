create schema LoTR;
use LoTR;

create table hero (
	hero_id int not null AUTO_INCREMENT,
	name char(125),
	hp int not null,
	strength int not null,
	level int not null,
	xp int,
	primary key (hero_id)	
);

create table enemy (
  enemy_id int not null AUTO_INCREMENT,
  name char(125),
  hp int not null,
  strength int not null,
  xp int not null,
  primary key (enemy_id)
);
